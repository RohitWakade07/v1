#!/bin/bash

# Define the base directory (in the sandbox, we'll use the current directory or $HOME)
BASE_DIR="$HOME/eep-software"

echo "1. Creating directory tree..."
mkdir -p "$BASE_DIR"
cd "$BASE_DIR" || exit 1
mkdir -p week-{01..12} notes scripts capstone

echo "2. Populating README.md files..."
for i in {01..12}; do
    echo "This is the README for week-$i." > "week-$i/README.md"
done

echo "3. Adding aliases to .bashrc..."
# The sandbox might not have a .bashrc, so we'll just create or append to it
echo "alias ll='ls -alF'" >> "$HOME/.bashrc"
echo "alias ..='cd ..'" >> "$HOME/.bashrc"

echo "4. Generating workspace-report.txt..."
REPORT_FILE="$BASE_DIR/workspace-report.txt"

echo "=== DIRECTORY TREE ===" > "$REPORT_FILE"
# Use ls -R because tree might not be installed in the alpine container
ls -R "$BASE_DIR" >> "$REPORT_FILE"

echo -e "\n=== .bashrc ADDITIONS ===" >> "$REPORT_FILE"
tail -n 2 "$HOME/.bashrc" >> "$REPORT_FILE"

echo "Mini project setup complete! Displaying the final report:"
cat "$REPORT_FILE"


sleep 100
