#!/bin/bash

echo "=== DOCKER VERIFICATION TEST ==="
echo ""

echo "1. Checking for Docker environment file (.dockerenv):"
if [ -f /.dockerenv ]; then
    echo "✅ /.dockerenv file exists! We are definitely inside a Docker container."
else
    echo "❌ /.dockerenv not found."
fi

echo -e "\n2. Checking root file system:"
ls -la / | head -n 5

echo -e "\n3. Checking process limits (Sandbox limits processes to 64):"
ulimit -a | grep "max user processes"

echo -e "\n4. Checking memory limits (Sandbox limits RAM to 128MB):"
free -m

echo -e "\n5. Checking Network Isolation (Sandbox disables networking):"
ping -c 1 8.8.8.8 || echo "✅ Network is unreachable (Sandbox network_mode is none!)"

echo -e "\n6. Checking OS Details:"
cat /etc/os-release
